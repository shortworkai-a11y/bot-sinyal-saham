'#3fb950' : '#f85149'; htmlContent += ` ${item.ticker} Rp ${item.price} ${item.change}% ${item.signal} `; }); tbody.innerHTML = htmlContent; document.getElementById('clock').innerText = new Date().toLocaleTimeString(); } catch (error) { console.error("Fetch error:", error); tbody.innerHTML = 'Koneksi Timeout.
